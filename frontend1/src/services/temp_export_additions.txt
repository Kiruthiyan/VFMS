    exportDriverReportPDF: (driverId: string) => {
        const driver = DRIVER_PERFORMANCE.find(d => d.driverId === driverId);
        if (!driver) {
            alert('Driver not found!');
            return;
        }
        
        const doc = new jsPDF();
        const fileName = Driver__Report.pdf;
        
        doc.setFontSize(20);
        doc.setFont('helvetica', 'bold');
        doc.text(Driver Report: , 14, 20);
        
        doc.setFontSize(11);
        doc.setFont('helvetica', 'normal');
        doc.text(Generated: , 14, 30);
        
        autoTable(doc, {
            startY: 40,
            head: [['Field', 'Value']],
            body: [
                ['Driver ID', driver.driverId],
                ['Name', driver.driverName],
                ['License Number', driver.licenseNumber],
                ['Assigned Vehicle', driver.assignedVehicle],
                ['Total Trips', driver.totalTrips.toString()],
                ['Total Distance (km)', driver.totalDistance.toLocaleString()],
                ['Violations', driver.violations.toString()],
                ['Feedback Rating', ${driver.feedbackRating}/5.0],
                ['Safety Score', ${driver.safetyScore}/100],
                ['On-Time Delivery Rate', ${driver.onTimeDeliveryRate}%]
            ],
            theme: 'grid',
            headStyles: { fillColor: [139, 92, 246] }
        });
        
        doc.save(fileName);
    },
    
    exportDriverReportExcel: (driverId: string) => {
        const driver = DRIVER_PERFORMANCE.find(d => d.driverId === driverId);
        if (!driver) {
            alert('Driver not found!');
            return;
        }
        
        const fileName = Driver__Report.xlsx;
        const data = [
            { Field: 'Driver ID', Value: driver.driverId },
            { Field: 'Name', Value: driver.driverName },
            { Field: 'License Number', Value: driver.licenseNumber },
            { Field: 'Assigned Vehicle', Value: driver.assignedVehicle },
            { Field: 'Total Trips', Value: driver.totalTrips },
            { Field: 'Total Distance (km)', Value: driver.totalDistance },
            { Field: 'Violations', Value: driver.violations },
            { Field: 'Feedback Rating', Value: driver.feedbackRating },
            { Field: 'Safety Score', Value: driver.safetyScore },
            { Field: 'On-Time Delivery Rate (%)', Value: driver.onTimeDeliveryRate }
        ];
        
        const ws = XLSX.utils.json_to_sheet(data);
        const wb = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(wb, ws, 'Driver Report');
        XLSX.writeFile(wb, fileName);
    },
    
    // Individual Fuel Log Report (by vehicle)
    exportFuelLogReportPDF: (vehicleId: string) => {
        const vehicle = VEHICLE_DETAILS.find(v => v.vehicleId === vehicleId);
        const fuelLogs = FUEL_LOGS.filter(f => f.vehicleId === vehicleId);
        
        if (!vehicle) {
            alert('Vehicle not found!');
            return;
        }
        
        const doc = new jsPDF();
        const fileName = Fuel_Log__Report.pdf;
        
        doc.setFontSize(20);
        doc.setFont('helvetica', 'bold');
        doc.text(Fuel Logs: , 14, 20);
        
        doc.setFontSize(11);
        doc.setFont('helvetica', 'normal');
        doc.text(Vehicle:  , 14, 30);
        doc.text(Total Logs: , 14, 36);
        
        const totalCost = fuelLogs.reduce((sum, f) => sum + f.totalCost, 0);
        const totalQuantity = fuelLogs.reduce((sum, f) => sum + f.fuelQuantity, 0);
        
        autoTable(doc, {
            startY: 44,
            head: [['Date', 'Quantity (L)', 'Price/L', 'Total Cost', 'Station']],
            body: fuelLogs.map(f => [
                f.date,
                f.fuelQuantity.toString(),
                $,
                $,
                f.fuelStation
            ]),
            foot: [['TOTAL', totalQuantity.toString() + ' L', '', $, '']],
            theme: 'grid',
            headStyles: { fillColor: [34, 197, 94] }
        });
        
        doc.save(fileName);
    },
